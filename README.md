```html
        Folder 1

        <u1-tree1>File 1.2</u1-tree1>
    </u1-tree1>
    <u1-tree1>
        Folder 2
        <u1-tree1>File 2.1</u1-tree1>
        <u1-tree1>File 2.2</u1-tree1>
    </u1-tree1>
    <u1-tree1>File 3</u1-tree1>
</u1-tree1>
```
